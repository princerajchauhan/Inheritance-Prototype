Q4. Write a JavaScript function to retrieve all the names of object's own and inherited properties. 


function propertyNames(obj) {
    var keys = [];
    for (var prop in obj) {
        if (obj.hasOwnProperty(prop)) {
            keys.push(prop);
        } 
    } 
    var proto = Object.getPrototypeOf(obj);
    if (proto !== null) {
        var inheritPropNames = propertyNames(proto);
        keys = keys.concat(inheritPropNames);
    } 
    return keys;
} 
function Person() {
    this.gender = "Male";
} 
Person.prototype = {
    name: "Prince",
    age: 21
};
var person = new Person();
person.city = "Delhi";
console.log(propertyNames(person));
Output :[ 'gender', 'city', 'name', 'age' ]
